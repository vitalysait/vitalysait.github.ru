#include <UniversalTelegramBot.h>
#include <DHT.h>

// ========== ТВОИ ДАННЫЕ ==========
const char* ssid = "ваш WiFI";
const char* password = "пароль от него";
#define BOT_TOKEN "8530454620:AAE-HtR87zyZ91dSYn9auueMH3eeoGard4o"

// ID разрешенных пользователей
const String allowedUsers[] = {
  "1234567890",
  "123456789",
  "1234567890"
};
const int totalUsers = 3;
// =================================

#define DHTPIN 2
#define DHTTYPE DHT22

DHT dht(DHTPIN, DHTTYPE);
WiFiClientSecure client;
UniversalTelegramBot bot(BOT_TOKEN, client);

float temperature = 0;
float humidity = 0;
unsigned long lastTime = 0;

bool isAllowed(String chat_id) {
  for(int i = 0; i < totalUsers; i++) {
    if(allowedUsers[i] == chat_id) return true;
  }
  return false;
}

void setup() {
  Serial.begin(115200);
  dht.begin();
  client.setInsecure();
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);
  bot.sendMessage("404415869", "꧁༺ *БОТ ЗАПУЩЕН* ༻꧂", "Markdown");
  bot.sendMessage("332677564", "꧁༺ *БОТ ЗАПУЩЕН* ༻꧂", "Markdown");
  bot.sendMessage("8473046860", "꧁༺ *БОТ ЗАПУЩЕН* ༻꧂", "Markdown");
}

void loop() {
  if(millis() - lastTime > 5000) {
    float t = dht.readTemperature();
    float h = dht.readHumidity();
    if(!isnan(t) && !isnan(h)) { temperature = t; humidity = h; }
    
    int msgCount = bot.getUpdates(bot.last_message_received + 1);
    for(int i = 0; i < msgCount; i++) {
      String chat = bot.messages[i].chat_id;
      String text = bot.messages[i].text;
      String name = bot.messages[i].from_name;
      
      if(!isAllowed(chat)) { bot.sendMessage(chat, "⛔️ *НЕТ ДОСТУПА*", "Markdown"); continue; }
      
      if(text == "/start" || text == "/menu") {
        String msg = "";
        String upperName = String(name);
        upperName.toUpperCase();
        msg += "꧁༺ *ДОБРО ПОЖАЛОВАТЬ* ༻꧂\n\n*ПРИВЕТ, " + upperName + "*\n";

      else if(text == "/temp" || text == "/TEMP") {
        String msg = "꧁༺ *ТЕМПЕРАТУРА* ༻꧂\n\n┏━━━━━━━━━━━━┓\n┃  *🌡  " + String(temperature, 1) + "°C*\n┃            ┃\n┗━━━━━━━━━━━━┛";

      }
      else if(text == "/hum" || text == "/HUM") {
        String msg = "꧁༺ *ВЛАЖНОСТЬ* ༻꧂\n\n┏━━━━━━━━━━━━┓\n┃  *💧  " + String(humidity, 1) + "%*\n┃            ┃\n┗━━━━━━━━━━━━┛";

        String msg = "꧁༺ *ВСЕ ДАННЫЕ* ༻꧂\n\n╔══════════════════╗\n║  *🌡 ТЕМПЕРАТУРА*\n║  *" + String(temperature, 1) + "°C*\n╠══════════════════╣\n║  *💧 ВЛАЖНОСТЬ*\n║  *" + String(humidity, 1) + "%*\n╚══════════════════╝\n\n┌──────────────┐\n│ ";

        else if(temperature < 10) msg += "*🌨 ХОЛОДНО*";
        else if(temperature < 20) msg += "*🌥 СВЕЖО*";
        else if(temperature < 30) msg += "*☀️ ТЕПЛО*";
        else msg += "*🔥 ЖАРА*";
        msg += "\n└──────────────┘";
        bot.sendMessage(chat, msg, "Markdown");
      }
      else if(text == "/style" || text == "/STYLE") {
        String msg = "꧁༺ *НАШ СТИЛЬ* ༻꧂\n\n┏━━━━━━━━━━━━━━━━┓\n┃ *꧁༺ ТЕКСТ ༻꧂*    ┃\n┃ *꧁༒ ПРИМЕР ༒꧂*   ┃\n┃ *꧁☂️︎シ︎✰*          ┃\n┃ *ঔৣ☬✞*           ┃\n┗━━━━━━━━━━━━━━━━┛\n\n*✨ КАК ИСПОЛЬЗОВАТЬ:*\n*СКОПИРУЙ ꧁༺ И ༻꧂*\n*И ВСТАВЬ СВОЙ ТЕКСТ*";

void drawSix(int x) {
  lcd.setCursor(x, 0); lcd.write(4); lcd.write(2); lcd.write(2);
  lcd.setCursor(x, 1); lcd.write(4); lcd.write(3); lcd.write(4);
}

void drawSeven(int x) {
  lcd.setCursor(x, 0); lcd.write(1); lcd.write(1); lcd.write(4);
  lcd.setCursor(x, 1); lcd.write(7); lcd.write(7); lcd.write(4);
}

  lcd.init();
  lcd.backlight();
  lcd.createChar(1, bar1);

  lcd.createChar(3, bar3);
  lcd.createChar(4, bar4);
  lcd.createChar(5, bar5);
  lcd.createChar(6, bar6);
  lcd.createChar(7, bar7);
  lcd.createChar(8, bar8);
  lcd.clear();
  drawSix(5);
  drawSeven(9);
}

Комплектация: В комплекте уже есть запасные лезвия, которые удобно хранить прямо в рукоятке — своего рода «багажник» всегда под рукой.

Вердикт: Очень доволен покупкой, рекомендую для резки картона, бумаги, пластика и обоев. Лезвия меняются легко, держатся долго. Для дома, гаража или мастерской — мастхэв.

Особенности: Особенно удобен для подготовки деталей перед пайкой или сваркой — убирает заусенцы, выравнивает края, делает кромки аккуратными и безопасными. Заменяет сразу несколько инструментов в мастерской.

Вердикт: Компактный, лёгкий, всегда под рукой. Для самодельщика, электрика, сантехника — просто незаменимая вещь. Качество сборки отличное, лезвия держат заточку долго. Рекомендую всем, кто любит мастерить!

Описание: Мощная турбозажигалка с ветрозащитой — настоящая находка для радиолюбителя и мастера. Отлично подходит для усадки термоусадочных трубок при пайке проводов, обжигания оплётки, удаления заусенцев на пластике.

Хозяйственное применение: Многоразовая, заправляется обычным газом для зажигалок. В хозяйстве пригодится не только для электроники, но и для розжига мангала, кулинарных работ (карамелизация), подпаливания ниток и многого другого.

Вердикт: Компактная, стильная, удобно лежит в руке. Очень доволен покупкой! Рекомендую для пайки и дома, и в поездках.
